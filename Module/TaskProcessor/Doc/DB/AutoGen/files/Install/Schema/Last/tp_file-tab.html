tp_file

create table tp_file (
   task_id integer not null ,
   file_status_code varchar2(10) not null ,
   file_name varchar2(100) not null ,
   mime_type_code varchar2(30) not null ,
   file_data clob ,
   loaded_date date ,
   date_ins date default sysdate not null ,
   operator_id integer not null ,
   constraint tp_file_pk primary key ( task_id) using index tablespace &indexTablespace ,
   constraint tp_file_ck_loaded_date check ( file_status_code = 'LOADING' and loaded_date is null or file_status_code != 'LOADING' and loaded_date is not null)
)
