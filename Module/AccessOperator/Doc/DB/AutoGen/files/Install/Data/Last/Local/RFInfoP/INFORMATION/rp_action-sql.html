create table rp_action (
   action_type_code varchar2( 10) not null ,
   database_link varchar2( 30) not null ,
   update_date date ,
   last_block_number integer ,
   error_code number ,
   error_message varchar2( 4000) ,
   error_date date ,
   customize_type integer default 0 ,
   date_ins date default sysdate not null ,
   operator_id integer not null ,
   constraint rp_action_PK primary key ( action_type_code, database_link) using index tablespace &indexTablespace ,
   constraint rp_action_ck_database_link check (nullif( database_link, upper( trim( database_link))) is null)
)
